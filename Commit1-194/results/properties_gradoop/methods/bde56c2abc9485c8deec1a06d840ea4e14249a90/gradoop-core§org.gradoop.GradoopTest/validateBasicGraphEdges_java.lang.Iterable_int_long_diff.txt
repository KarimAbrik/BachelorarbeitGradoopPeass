private void validateBasicGraphEdges(Iterable&lt;Edge&gt; edges, int expectedCount, long... otherIDs |                                                                                                     
List&lt;Edge&gt; edgeList = Lists.newArrayList(edges);                                               |                                                                                                     
assertThat(edgeList.size(), is(expectedCount));                                                      |                                                                                                     
for (int i = 0; i &lt; otherIDs.length; i++) {                                                       |                                                                                                     
boolean match = false;                                                                               |                                                                                                     
for (Edge e : edgeList) {                                                                            |                                                                                                     
if (e.getOtherID() == otherIDs[i]) {                                                                 |                                                                                                     
match = true;                                                                                        |                                                                                                     
assertThat(e.getIndex(), is((long) i));                                                              |                                                                                                     
}                                                                                                    |                                                                                                     
}                                                                                                    |                                                                                                     
if (!match) {                                                                                        |                                                                                                     
assertTrue("edge list contains wrong edges", false);                                                 |                                                                                                     
}                                                                                                    |                                                                                                     
}                                                                                                    |                                                                                                     
}                                                                                                    |
