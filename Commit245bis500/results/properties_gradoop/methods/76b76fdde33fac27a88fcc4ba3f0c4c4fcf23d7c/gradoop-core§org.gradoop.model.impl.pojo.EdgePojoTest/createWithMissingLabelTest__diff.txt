@Test
public void createWithMissingLabelTest() {
    GradoopIdGenerator idGen = new SequenceIdGenerator();
    GradoopId edgeId = idGen.createId();
    GradoopId sourceId = idGen.createId();
    GradoopId targetId = idGen.createId();
