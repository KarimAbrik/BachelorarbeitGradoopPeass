/**
 * Creates a new Configuration.
 *
 * @param vertexHandler     vertex handler
 * @param edgeHandler       edge handler
 * @param graphHeadHandler      graph head handler
 * @param executionEnvironment  Flink execution environment
 */
private GradoopFlinkConfig(VertexHandler<VD, ED> vertexHandler, EdgeHandler<ED, VD> edgeHandler, GraphHeadHandler<GD> graphHeadHandler, ExecutionEnvironment executionEnvironment) {
    super(vertexHandler, edgeHandler, graphHeadHandler);
    if (executionEnvironment == null) {
        throw new IllegalArgumentException("Execution environment must not be null");
    }
    this.executionEnvironment = executionEnvironment;
}
