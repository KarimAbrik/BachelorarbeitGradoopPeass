@Override
public EPGMEdge map(String csvLine) throws Exception {
    String[] tokens = split(csvLine, 6);
    String label = StringEscaper.unescape(tokens[4]);
    return edgeFactory.initEdge(GradoopId.fromString(tokens[0]), label, GradoopId.fromString(tokens[2]), GradoopId.fromString(tokens[3]), parseProperties(MetaDataSource.EDGE_TYPE, label, tokens[5]), parseGradoopIds(tokens[1]));
}
