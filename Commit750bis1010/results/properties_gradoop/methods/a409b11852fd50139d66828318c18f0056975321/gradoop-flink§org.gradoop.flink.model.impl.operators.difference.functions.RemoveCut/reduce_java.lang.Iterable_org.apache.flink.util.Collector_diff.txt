@Override
public void reduce(Iterable<Tuple2<O, Long>> iterable, Collector<O> collector) throws Exception {
    boolean inFirst = false;
    boolean inSecond = false;
    O o = null;
    for (Tuple2<O, Long> tuple : iterable) {
        o = tuple.f0;
        if (tuple.f1 == 1L) {
            inFirst = true;
        } else {
            inSecond = true;
        }
    }
    if (inFirst && !inSecond) {
        collector.collect(o);
    }
}
