@Test
public void testNonDistinctCollection() throws Exception {
    FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
    GraphCollection inputCollection = loader.getGraphCollectionByVariables("g0", "g0", "g1");
    GraphCollection expectedCollection = loader.getGraphCollectionByVariables("g0", "g1");
    GraphCollection outputCollection = inputCollection.distinctById();
    collectAndAssertTrue(outputCollection.equalsByGraphElementIds(expectedCollection));
}
