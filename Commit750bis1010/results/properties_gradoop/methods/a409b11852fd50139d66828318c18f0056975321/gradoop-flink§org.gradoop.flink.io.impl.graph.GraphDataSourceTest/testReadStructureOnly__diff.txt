@Test
public void testReadStructureOnly() throws Exception {
    ExecutionEnvironment env = getExecutionEnvironment();
    DataSet<ImportVertex<Long>> importVertices = env.fromElements(new ImportVertex<>(0L), new ImportVertex<>(1L));
    DataSet<ImportEdge<Long>> importEdges = env.fromElements(new ImportEdge<>(0L, 0L, 1L));
    LogicalGraph expected = getLoaderFromString("expected[()-->()]").getLogicalGraphByVariable("expected");
    GraphDataSource<Long> dataSource = new GraphDataSource<>(importVertices, importEdges, getConfig());
    LogicalGraph output = dataSource.getLogicalGraph();
    collectAndAssertTrue(output.equalsByElementData(expected));
}
