@Override
protected String getDebugString(Embedding embedding) {
    String vertexMapping = embeddingMetaData.getVertexVariables().stream().map(var -> String.format("%s : %s", var, vertexMap.get(embedding.getId(embeddingMetaData.getEntryColumn(var))))).collect(Collectors.joining(", "));
    String edgeMapping = embeddingMetaData.getEdgeVariables().stream().map(var -> String.format("%s : %s", var, edgeMap.get(embedding.getId(embeddingMetaData.getEntryColumn(var))))).collect(Collectors.joining(", "));
    String pathMapping = embeddingMetaData.getPathVariables().stream().map(var -> {
        List<GradoopId> path = embedding.getIdList(embeddingMetaData.getEntryColumn(var));
        List<PropertyValue> ids = new ArrayList<>();
        for (int i = 0; i < path.size(); i++) {
            if (i % 2 == 0) {
                // edge
                ids.add(edgeMap.get(path.get(i)));
            } else {
                ids.add(vertexMap.get(path.get(i)));
            }
        }
        return String.format("%s : %s", var, ids);
    }).collect(Collectors.joining(", "));
    return String.format("vertex-mapping: {%s}, edge-mapping: {%s}, path-mapping: {%s}", vertexMapping, edgeMapping, pathMapping);
}
