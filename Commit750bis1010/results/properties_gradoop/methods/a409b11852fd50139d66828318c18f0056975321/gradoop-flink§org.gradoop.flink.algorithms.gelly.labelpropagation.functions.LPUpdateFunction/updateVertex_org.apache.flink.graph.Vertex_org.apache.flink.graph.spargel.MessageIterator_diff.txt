/**
 * Updates the vertex value if it has changed.
 *
 * @param vertex  vertex to be updated
 * @param msg     message
 */
@Override
public void updateVertex(Vertex<GradoopId, PropertyValue> vertex, MessageIterator<PropertyValue> msg) {
    PropertyValue value = getNewValue(vertex, Lists.newArrayList(msg.iterator()));
    if (!vertex.getValue().equals(value)) {
        setNewVertexValue(value);
    }
}
