@Test
public void testEdgeConstructionReducedPattern() throws Exception {
    FlinkAsciiGraphLoader loader = getSocialNetworkLoader();
    LogicalGraph dbGraph = loader.getLogicalGraph();
    loader.appendToDatabaseFromString("expected0[" + "(alice)-[:possible_friend]->(carol)" + "]," + "expected1[" + "(bob)-[:possible_friend]->(dave)" + "]");
    GraphCollection result = dbGraph.query("MATCH (a:Person)-[:knows]->(b:Person)-[:knows]->(c:Person) " + "WHERE a.city = 'Leipzig' AND a <> c", "(a)-[e_new:possible_friend]->(c)");
    GraphCollection expectedCollection = loader.getGraphCollectionByVariables("expected0", "expected1");
    collectAndAssertTrue(result.equalsByGraphElementData(expectedCollection));
}
