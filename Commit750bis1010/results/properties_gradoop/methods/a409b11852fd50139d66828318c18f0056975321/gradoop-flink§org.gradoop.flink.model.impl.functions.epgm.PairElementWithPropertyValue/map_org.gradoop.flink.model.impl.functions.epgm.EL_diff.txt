@Override
public Tuple2<GradoopId, PropertyValue> map(EL el) throws Exception {
    reuseTuple.f0 = el.getId();
    if (el.hasProperty(propertyKey)) {
        reuseTuple.f1 = el.getPropertyValue(propertyKey);
    } else {
        reuseTuple.f1 = PropertyValue.NULL_VALUE;
    }
    return reuseTuple;
}
