@Override
public boolean equals(Object o) {
    if (this == o) {
        return true;
    }
    if (o == null || getClass() != o.getClass()) {
        return false;
    }
    Step step = (Step) o;
    if (from != step.from) {
        return false;
    }
    if (via != step.via) {
        return false;
    }
    if (to != step.to) {
        return false;
    }
    return isOutgoing == step.isOutgoing;
}
