@Override
public EPGMEdge join(EPGMEdge first, EPGMVertex second) throws Exception {
    first.setProperty(ImageDataSink.SOURCE_X, second.getPropertyValue(LayoutingAlgorithm.X_COORDINATE_PROPERTY));
    first.setProperty(ImageDataSink.SOURCE_Y, second.getPropertyValue(LayoutingAlgorithm.Y_COORDINATE_PROPERTY));
    return first;
}
