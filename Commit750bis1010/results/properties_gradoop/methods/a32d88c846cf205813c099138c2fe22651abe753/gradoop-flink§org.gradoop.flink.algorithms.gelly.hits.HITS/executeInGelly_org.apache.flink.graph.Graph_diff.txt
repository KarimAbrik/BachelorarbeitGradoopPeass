@Override
public LogicalGraph executeInGelly(Graph<GradoopId, NullValue, NullValue> graph) throws Exception {
    DataSet<EPGMVertex> newVertices = hits.runInternal(graph).join(currentGraph.getVertices()).where(new HitsResultKeySelector()).equalTo(new Id<>()).with(new HITSToAttributes(authorityPropertyKey, hubPropertyKey));
    return currentGraph.getFactory().fromDataSets(currentGraph.getGraphHead(), newVertices, currentGraph.getEdges());
}
