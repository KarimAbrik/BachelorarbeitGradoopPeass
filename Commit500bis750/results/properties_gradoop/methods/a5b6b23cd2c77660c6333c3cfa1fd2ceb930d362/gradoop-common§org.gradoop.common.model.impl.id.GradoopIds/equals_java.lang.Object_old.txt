@Override
public boolean equals(Object o) {
    boolean equal = this == o;
    if (!equal && o instanceof GradoopIds) {
        GradoopIds that = (GradoopIds) o;
        // same number of ids
        equal = this.size() == that.size();
        if (equal) {
            // same ids
            equal = this.ids.equals(that.ids);
        }
    }
    return equal;
}
