*                                                                                                    |

* @return true, iff any of the specified ids is contained in the set                                 |                                                                                                     
*/                                                                                                   |

for (GradoopId id : ids) {                                                                           |                                                                                                     
if (this.contains(id)) {                                                                             |                                                                                                     
return true;                                                                                         |                                                                                                     
}                                                                                                    |                                                                                                     
}                                                                                                    |                                                                                                     
return false;                                                                                        |                                                                                                     
}                                                                                                    |
