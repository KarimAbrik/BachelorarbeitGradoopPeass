@Test
public void testMapValueMaxSize() {
    Map<PropertyValue, PropertyValue> m = new HashMap<>();
    // 8 bytes per double + 1 byte overhead
    for (int i = 0; i < PropertyValue.MAX_BINARY_LENGTH / 18; i++) {
        PropertyValue p = create(Math.random());
        m.put(p, p);
    }
    create(m);
}
