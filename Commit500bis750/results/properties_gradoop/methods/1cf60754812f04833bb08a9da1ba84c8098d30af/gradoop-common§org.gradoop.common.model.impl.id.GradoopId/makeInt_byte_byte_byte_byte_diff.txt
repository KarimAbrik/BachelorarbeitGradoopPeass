/**
 * Returns a primitive int represented by the given 4 bytes.
 *
 * @param b3 byte 3
 * @param b2 byte 2
 * @param b1 byte 1
 * @param b0 byte 0
 *
 * @return int value
 */
private static int makeInt(final byte b3, final byte b2, final byte b1, final byte b0) {
    return (b3 << 24) | ((b2 & 0xff) << 16) | ((b1 & 0xff) << 8) | ((b0 & 0xff));
}
